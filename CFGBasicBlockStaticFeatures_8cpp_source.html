200 void CFGBasicBlockStaticFeatures::processModule(llvm::Module& Module) {
201  if (this->isProcessed()) return;
202  Processed = true;
203 
204  uint64_t NamelessCount = 0;
205  for (auto &Function : Module) {
206  uint64_t Count = 0;
207  std::string FunctionName = Function.getName();
208  if (FunctionName == "") FunctionName = "Nameless" + std::to_string(NamelessCount++);
209  for (auto &BasicBlock : Function) {
