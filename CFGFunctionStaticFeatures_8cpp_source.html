65 void CFGFunctionStaticFeatures::processBasicBlockOfFunction(llvm::BasicBlock &BasicBlock, std::string FunctionName) {
66  int NumPredecessors = 0, NumSuccessors = 0;
67 
68  NumSuccessors = BasicBlock.getTerminator()->getNumSuccessors();
69  for (auto I = llvm::pred_begin(&BasicBlock), E = llvm::pred_end(&BasicBlock); I != E; ++I)
70  ++NumPredecessors;
71 
72  addOneToSubFeatureOfKey("nof_bb", FunctionName);
73 
74  switch(NumPredecessors) {
75  case 0: break;
76  case 1: addOneToSubFeatureOfKey("nof_1pred_bb", FunctionName);
77  case 2: addOneToSubFeatureOfKey("nof_2pred_bb", FunctionName);
78  default: addOneToSubFeatureOfKey("nof_g2pred_bb", FunctionName);
79  }
80 
81  switch(NumSuccessors) {
82  case 0: break;
83  case 1: addOneToSubFeatureOfKey("nof_1suc_bb", FunctionName);
84  case 2: addOneToSubFeatureOfKey("nof_2suc_bb", FunctionName);
85  default: addOneToSubFeatureOfKey("nof_g2suc_bb", FunctionName);
86  }
87 
88  if (NumSuccessors == 1 && NumPredecessors == 1)
89  addOneToSubFeatureOfKey("nof_1pred_1suc_bb", FunctionName);
90  else if (NumSuccessors == 1 && NumPredecessors == 2)
91  addOneToSubFeatureOfKey("nof_1pred_2suc_bb", FunctionName);
92  else if (NumSuccessors == 2 && NumPredecessors == 1)
93  addOneToSubFeatureOfKey("nof_2pred_1suc_bb", FunctionName);
94  else if (NumSuccessors == 2 && NumPredecessors == 2)
95  addOneToSubFeatureOfKey("nof_2pred_2suc_bb", FunctionName);
96  else if (NumSuccessors > 2 && NumPredecessors > 2)
97  addOneToSubFeatureOfKey("nof_g2pred_g2suc_bb", FunctionName);
98 
99  int TotalEdges = NumSuccessors + getValueOfKey("nof_cfg_edges", FunctionName);
100  setValueOfKey("nof_cfg_edges", TotalEdges, FunctionName);
101 
102  for (auto I = 0; I < NumSuccessors; ++I)
103  if (isCriticalEdge(BasicBlock.getTerminator(), I, true))
104  addOneToSubFeatureOfKey("nof_cfg_crit_edges", FunctionName);
105
